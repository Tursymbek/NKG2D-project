# =========================================================
# TCGA + GTEx NKG2D ligand score across tissues
# FULL SCRIPT: Cell Press colors + hidden outliers + log1p scale
# =========================================================

suppressPackageStartupMessages({
    library(data.table)
    library(dplyr)
    library(ggplot2)
    library(org.Hs.eg.db)
    library(AnnotationDbi)
})

# =========================================================
# COLORS — Cell Press style
# =========================================================
col_normal <- "#3F8FBF"
col_tumor  <- "#C94F4F"

# =========================================================
# 1. FILES
# =========================================================
expr_file  <- "TcgaTargetGtex_filtered.tsv"
pheno_file <- "TcgaTargetGTEX_phenotype.txt"

# =========================================================
# 2. LOAD DATA
# =========================================================
expr  <- fread(expr_file, sep = "\t", data.table = FALSE)
pheno <- fread(pheno_file, data.table = FALSE)

cat("Expression dim:", dim(expr), "\n")
cat("Phenotype dim :", dim(pheno), "\n")

# =========================================================
# 3. FIX EXPRESSION TABLE
# =========================================================
colnames(expr)[1] <- "EnsemblID"
expr$EnsemblID <- gsub("\\..*$", "", expr$EnsemblID)

expr$GeneSymbol <- mapIds(
    x = org.Hs.eg.db,
    keys = expr$EnsemblID,
    column = "SYMBOL",
    keytype = "ENSEMBL",
    multiVals = "first"
)

expr <- expr[!is.na(expr$GeneSymbol) & expr$GeneSymbol != "", , drop = FALSE]

cat("After Ensembl->SYMBOL mapping:", nrow(expr), "genes\n")

# =========================================================
# 4. SELECT NKG2D LIGANDS
# ULBP4/5/6 may appear as RAET1E/G/L
# =========================================================
ligand_map <- c(
    "MICA"   = "MICA",
    "MICB"   = "MICB",
    "ULBP1"  = "ULBP1",
    "ULBP2"  = "ULBP2",
    "ULBP3"  = "ULBP3",
    "RAET1E" = "ULBP4",
    "ULBP4"  = "ULBP4",
    "RAET1G" = "ULBP5",
    "ULBP5"  = "ULBP5",
    "RAET1L" = "ULBP6",
    "ULBP6"  = "ULBP6"
)

expr_sub <- expr[expr$GeneSymbol %in% names(ligand_map), , drop = FALSE]

cat("Rows matching NKG2DL genes before collapsing duplicates:", nrow(expr_sub), "\n")
print(sort(unique(expr_sub$GeneSymbol)))

if (nrow(expr_sub) == 0) {
    stop("No NKG2DL genes found after Ensembl->SYMBOL mapping.")
}

expr_sub$Ligand <- unname(ligand_map[expr_sub$GeneSymbol])

num_cols <- vapply(expr_sub, is.numeric, logical(1))
expr_sub2 <- expr_sub[, c("Ligand", names(expr_sub)[num_cols]), drop = FALSE]

expr_sub2 <- expr_sub2 %>%
    dplyr::group_by(Ligand) %>%
    dplyr::summarise(
        dplyr::across(dplyr::everything(), ~mean(.x, na.rm = TRUE)),
        .groups = "drop"
    )

cat("Rows after collapsing duplicated ligands:", nrow(expr_sub2), "\n")
print(expr_sub2$Ligand)

# =========================================================
# 5. CALCULATE NKG2DL SCORE
# Xena values are log2(tpm + 0.001), convert back to linear TPM
# =========================================================
expr_mat <- as.matrix(expr_sub2[, -1, drop = FALSE])
rownames(expr_mat) <- expr_sub2$Ligand

expr_linear <- 2^expr_mat - 0.001
expr_linear[expr_linear < 0] <- 0

score_df <- data.frame(
    sample = colnames(expr_linear),
    NKG2DL_score = colSums(expr_linear, na.rm = TRUE),
    stringsAsFactors = FALSE
)

cat("Score table dim:", dim(score_df), "\n")

# =========================================================
# 6. FIX PHENOTYPE TABLE
# =========================================================
colnames(pheno) <- gsub("^_+", "", colnames(pheno))
colnames(pheno) <- gsub(" ", "_", colnames(pheno))
colnames(pheno) <- make.names(colnames(pheno), unique = TRUE)

sample_col <- intersect(c("sampleID", "sample", "SampleID"), colnames(pheno))[1]
if (is.na(sample_col)) stop("Could not find sample column in phenotype table.")

pheno$sample <- pheno[[sample_col]]

need_cols <- c(
    "sample", "detailed_category", "primary_disease_or_tissue",
    "primary_site", "sample_type", "gender", "study"
)

need_cols <- intersect(need_cols, colnames(pheno))
pheno2 <- pheno[, need_cols, drop = FALSE]

# =========================================================
# 7. MERGE
# =========================================================
df <- merge(score_df, pheno2, by = "sample")

cat("Merged dim:", dim(df), "\n")
print(table(df$study, useNA = "ifany"))
print(table(df$sample_type, useNA = "ifany"))

# =========================================================
# 8. DEFINE GROUPS
# =========================================================
df$group <- NA_character_

df$group[df$study == "TCGA" & df$sample_type == "Primary Tumor"] <- "Tumor"
df$group[df$study == "GTEX" & df$sample_type == "Normal Tissue"] <- "Normal"
df$group[df$study == "TCGA" & df$sample_type == "Solid Tissue Normal"] <- "Normal"

df2 <- df[!is.na(df$group), , drop = FALSE]

cat("After group assignment:", nrow(df2), "\n")
print(table(df2$group, useNA = "ifany"))

# =========================================================
# 9. MANUAL TISSUE MAPPING
# =========================================================
df2$tissue <- trimws(as.character(df2$primary_disease_or_tissue))
df2$tissue_simple <- NA_character_

df2$tissue_simple[grepl("Colon", df2$tissue, ignore.case = TRUE)] <- "Colon"
df2$tissue_simple[grepl("Rectum", df2$tissue, ignore.case = TRUE)] <- "Rectum"
df2$tissue_simple[grepl("Lung", df2$tissue, ignore.case = TRUE)] <- "Lung"
df2$tissue_simple[grepl("Breast", df2$tissue, ignore.case = TRUE)] <- "Breast"
df2$tissue_simple[grepl("Liver", df2$tissue, ignore.case = TRUE)] <- "Liver"
df2$tissue_simple[grepl("Kidney", df2$tissue, ignore.case = TRUE)] <- "Kidney"
df2$tissue_simple[grepl("Stomach", df2$tissue, ignore.case = TRUE)] <- "Stomach"
df2$tissue_simple[grepl("Thyroid", df2$tissue, ignore.case = TRUE)] <- "Thyroid"
df2$tissue_simple[grepl("Prostate", df2$tissue, ignore.case = TRUE)] <- "Prostate"
df2$tissue_simple[grepl("Skin", df2$tissue, ignore.case = TRUE)] <- "Skin"
df2$tissue_simple[grepl("Brain", df2$tissue, ignore.case = TRUE)] <- "Brain"
df2$tissue_simple[grepl("Bladder", df2$tissue, ignore.case = TRUE)] <- "Bladder"
df2$tissue_simple[grepl("Head", df2$tissue, ignore.case = TRUE)] <- "HeadNeck"
df2$tissue_simple[grepl("Blood|Leukemia|Bone Marrow", df2$tissue, ignore.case = TRUE)] <- "Blood"
df2$tissue_simple[grepl("Ovar", df2$tissue, ignore.case = TRUE)] <- "Ovary"
df2$tissue_simple[grepl("Uter|Endomet", df2$tissue, ignore.case = TRUE)] <- "Uterus"
df2$tissue_simple[grepl("Cerv", df2$tissue, ignore.case = TRUE)] <- "Cervix"
df2$tissue_simple[grepl("Esoph", df2$tissue, ignore.case = TRUE)] <- "Esophagus"
df2$tissue_simple[grepl("Pancrea", df2$tissue, ignore.case = TRUE)] <- "Pancreas"
df2$tissue_simple[grepl("Adipose", df2$tissue, ignore.case = TRUE)] <- "Adipose"
df2$tissue_simple[grepl("Muscle", df2$tissue, ignore.case = TRUE)] <- "Muscle"
df2$tissue_simple[grepl("Testis", df2$tissue, ignore.case = TRUE)] <- "Testis"
df2$tissue_simple[grepl("Spleen", df2$tissue, ignore.case = TRUE)] <- "Spleen"
df2$tissue_simple[grepl("Intestine", df2$tissue, ignore.case = TRUE)] <- "Intestine"

df_plot <- df2[!is.na(df2$tissue_simple), , drop = FALSE]

keep_tissues <- df_plot %>%
    dplyr::group_by(tissue_simple) %>%
    dplyr::summarise(
        has_tumor = any(group == "Tumor"),
        has_normal = any(group == "Normal"),
        .groups = "drop"
    ) %>%
    dplyr::filter(has_tumor & has_normal) %>%
    dplyr::pull(tissue_simple)

df_plot <- df_plot %>%
    dplyr::filter(tissue_simple %in% keep_tissues)

cat("Final tissues:\n")
print(sort(table(df_plot$tissue_simple), decreasing = TRUE))
print(table(df_plot$group))

if (nrow(df_plot) == 0) {
    stop("No tissues left after matching Tumor and Normal groups.")
}

# =========================================================
# 10. ORDER TISSUES BY DELTA MEDIAN
# =========================================================
ord_df <- df_plot %>%
    dplyr::group_by(tissue_simple, group) %>%
    dplyr::summarise(
        med = median(NKG2DL_score, na.rm = TRUE),
        .groups = "drop"
    )

ord_tumor  <- ord_df[ord_df$group == "Tumor",  c("tissue_simple", "med")]
ord_normal <- ord_df[ord_df$group == "Normal", c("tissue_simple", "med")]

ord_wide <- merge(
    ord_tumor,
    ord_normal,
    by = "tissue_simple",
    suffixes = c("_Tumor", "_Normal")
)

ord_wide$delta <- ord_wide$med_Tumor - ord_wide$med_Normal
ord_wide <- ord_wide[order(ord_wide$delta, decreasing = TRUE), ]

df_plot$tissue_simple <- factor(df_plot$tissue_simple, levels = ord_wide$tissue_simple)
df_plot$group <- factor(df_plot$group, levels = c("Normal", "Tumor"))

# =========================================================
# 11. STATS PER TISSUE
# IMPORTANT: statistics use full df_plot, outliers included
# =========================================================
stats_df <- lapply(split(df_plot, df_plot$tissue_simple), function(x) {
    
    if (length(unique(x$group)) < 2) return(NULL)
    
    wt <- wilcox.test(NKG2DL_score ~ group, data = x)
    
    ymax <- max(x$NKG2DL_score, na.rm = TRUE)
    ymin <- min(x$NKG2DL_score, na.rm = TRUE)
    yrng <- ymax - ymin
    if (!is.finite(yrng) || yrng == 0) yrng <- max(1, ymax * 0.1)
    
    data.frame(
        tissue_simple = unique(as.character(x$tissue_simple)),
        p_value = wt$p.value,
        median_normal = median(x$NKG2DL_score[x$group == "Normal"], na.rm = TRUE),
        median_tumor  = median(x$NKG2DL_score[x$group == "Tumor"],  na.rm = TRUE),
        delta_median  = median(x$NKG2DL_score[x$group == "Tumor"],  na.rm = TRUE) -
            median(x$NKG2DL_score[x$group == "Normal"], na.rm = TRUE),
        y_pos = ymax + 0.08 * yrng,
        y_seg = ymax + 0.03 * yrng
    )
})

stats_df <- do.call(rbind, stats_df)
stats_df$FDR <- p.adjust(stats_df$p_value, method = "BH")

stats_df$label_fdr <- paste0("FDR=", signif(stats_df$FDR, 3))
stats_df$stars <- ifelse(
    stats_df$FDR < 0.001, "***",
    ifelse(
        stats_df$FDR < 0.01, "**",
        ifelse(stats_df$FDR < 0.05, "*", "ns")
    )
)

stats_df <- stats_df[match(levels(df_plot$tissue_simple), stats_df$tissue_simple), ]

write.csv(stats_df, "NKG2DL_tissue_statistics_with_labels.csv", row.names = FALSE)

# =========================================================
# 12. MAIN PLOT — hidden outliers + log1p scale
# =========================================================
p_all_stars <- ggplot(df_plot, aes(x = tissue_simple, y = NKG2DL_score, fill = group)) +
    geom_boxplot(
        outlier.shape = NA,
        position = position_dodge(width = 0.75),
        linewidth = 0.45
    ) +
    geom_segment(
        data = stats_df,
        aes(
            x = as.numeric(factor(tissue_simple, levels = levels(df_plot$tissue_simple))) - 0.22,
            xend = as.numeric(factor(tissue_simple, levels = levels(df_plot$tissue_simple))) + 0.22,
            y = y_seg,
            yend = y_seg
        ),
        inherit.aes = FALSE,
        linewidth = 0.35
    ) +
    geom_text(
        data = stats_df,
        aes(
            x = tissue_simple,
            y = y_pos,
            label = stars
        ),
        inherit.aes = FALSE,
        size = 8,
        vjust = 0
    ) +
    scale_fill_manual(values = c("Normal" = col_normal, "Tumor" = col_tumor)) +
    scale_y_continuous(
        trans = "log1p",
        expand = expansion(mult = c(0.03, 0.22))
    ) +
    theme_classic(base_size = 20) +
    theme(
        plot.title = element_text(size = 28, face = "bold"),
        axis.title = element_text(size = 24),
        axis.text.x = element_text(size = 17, angle = 90, hjust = 1, vjust = 0.5),
        axis.text.y = element_text(size = 17),
        legend.text = element_text(size = 24),
        legend.title = element_blank(),
        legend.position = "top",
        axis.line = element_line(linewidth = 0.8),
        panel.grid = element_blank()
    ) +
    labs(
        title = "NKG2D ligand score across tumor and normal tissues",
        x = "Tissue",
        y = "NKG2DL score (log1p scale)"
    ) +
    coord_cartesian(clip = "off")

ggsave(
    "NKG2DL_boxplot_all_tissues_CellPress_hidden_outliers_log1p.png",
    p_all_stars,
    width = 16,
    height = 8,
    dpi = 300
)

# =========================================================
# 12B. BARPLOT — Tumor vs Normal median difference
# =========================================================

bar_df <- stats_df %>%
    dplyr::arrange(desc(delta_median))

bar_df$tissue_simple <- factor(bar_df$tissue_simple, levels = bar_df$tissue_simple)

p_delta <- ggplot(bar_df, aes(x = tissue_simple, y = delta_median)) +
    geom_hline(yintercept = 0, linewidth = 0.6, color = "#1F5F7A") +
    geom_col(fill = "#3A9FD1", width = 0.8) +
    geom_text(
        aes(
            label = stars,
            y = ifelse(delta_median >= 0, delta_median + 1.2, delta_median - 1.2)
        ),
        size = 8,
        vjust = ifelse(bar_df$delta_median >= 0, 0, 1)
    ) +
    theme_classic(base_size = 18) +
    theme(
        plot.title = element_text(size = 20, hjust = 0.5),
        axis.title = element_text(size = 18),
        axis.text.x = element_text(size = 18, angle = 45, hjust = 1),
        axis.text.y = element_text(size = 18),
        panel.grid = element_blank()
    ) +
    labs(
        title = "Tumor-versus-normal NKG2DL expression contrast",
        x = "Tissue type",
        y = "Tumor-Normal median difference\n(NKG2DL score)"
    ) +
    coord_cartesian(clip = "off")

ggsave(
    "NKG2DL_tumor_normal_median_difference_barplot.png",
    p_delta,
    width = 12,
    height = 6,
    dpi = 300
)

# =========================================================
# 13. SAVE FINAL TABLE
# =========================================================
write.csv(df_plot, "NKG2DL_per_sample_mapped.csv", row.names = FALSE)

cat("\nDone.\n")
cat("Outliers were hidden only in visualization using outlier.shape = NA.\n")
cat("Statistics were calculated using full unfiltered df_plot.\n")
cat("Main plot uses log1p scale.\n")
